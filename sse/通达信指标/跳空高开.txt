-- 跳空高开低开指标

UPQK:=LOW-REF(HIGH,1);
STICKLINE(CLOSE>REF(CLOSE,1) AND UPQK>0,REF(HIGH,1),LOW,2,1),COLORFF00FF;
DRAWTEXT(CLOSE>REF(CLOSE,1) AND UPQK>0,UPQK/2+REF(HIGH,1), 'XS'),COLORFF00FF;
DOWNQK:=REF(LOW,1)-HIGH;
STICKLINE(CLOSE<REF(CLOSE,1) AND DOWNQK>0,REF(LOW,1),HIGH,2,1),COLORFFFF00;
DRAWTEXT(CLOSE<REF(CLOSE,1) AND DOWNQK>0,DOWNQK/2+HIGH, 'XX'),COLORFFFF00;

STICKLINE(WEEKDAY=1) AND UPQK>0,REF(HIGH,1),LOW,2,1),COLORFF00FF;
